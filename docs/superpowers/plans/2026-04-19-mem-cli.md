```python
import re

FRONTMATTER_RE = re.compile(r"^---\n(.*?)\n---\n(.*)$", re.DOTALL)

def parse_frontmatter(text):
    """Parse '---\\n...\\n---\\n' YAML-like frontmatter. Returns (dict, body)."""
    m = FRONTMATTER_RE.match(text)
    if not m:
        return {}, text
    fm_raw, body = m.group(1), m.group(2)
    fm = {}
    for line in fm_raw.splitlines():
        if ":" not in line:
            continue
        key, _, val = line.partition(":")
        key = key.strip()
        val = val.strip()
        if val.startswith("[") and val.endswith("]"):
            inner = val[1:-1].strip()
            fm[key] = [x.strip() for x in inner.split(",") if x.strip()] if inner else []
        else:
            fm[key] = val
    return fm, body
```

```python
from pathlib import Path

EXCLUDED_DIRS = {".git", ".venv", "node_modules", "__pycache__", ".vscode",
                 ".claude", ".octogent", "dist", "build", ".sdd", ".atl"}

def scan(root):
    """Yield dicts {name, path, frontmatter, body, todo_path} for each PROJECT.md under root."""
    root = Path(root)
    for project_md in root.rglob("PROJECT.md"):
        if any(part in EXCLUDED_DIRS for part in project_md.parts):
            continue
        try:
            text = project_md.read_text(encoding="utf-8")
        except (OSError, UnicodeDecodeError):
            continue
        fm, body = parse_frontmatter(text)
        if not fm:
            continue
        name = fm.get("name") or project_md.parent.name
        yield {
            "name": name,
            "path": project_md.parent,
            "frontmatter": fm,
            "body": body,
            "todo_path": project_md.parent / "todo.md",
        }
```

```python
import argparse

def cmd_ls(argv, root):
    """List projects. Flags: --status, --tag."""
    p = argparse.ArgumentParser(prog="mem ls")
    p.add_argument("--status")
    p.add_argument("--tag")
    args = p.parse_args(argv)

    projects = list(scan(root))
    if args.status:
        projects = [x for x in projects if x["frontmatter"].get("status") == args.status]
    if args.tag:
        projects = [x for x in projects if args.tag in x["frontmatter"].get("tags", [])]

    if not projects:
        print("no projects found")
        return 0

    rows = [(x["name"],
             x["frontmatter"].get("status", "?"),
             ",".join(x["frontmatter"].get("tags", [])) or "-",
             x["frontmatter"].get("updated", "?")) for x in projects]
    rows.sort(key=lambda r: r[0].lower())
    widths = [max(len(r[i]) for r in rows + [("NAME", "STATUS", "TAGS", "UPDATED")]) for i in range(4)]
    header = ("NAME", "STATUS", "TAGS", "UPDATED")
    print("  ".join(h.ljust(widths[i]) for i, h in enumerate(header)))
    for r in rows:
        print("  ".join(r[i].ljust(widths[i]) for i in range(4)))
    return 0
```

```python
TODO_RE = re.compile(r"^\s*-\s*\[\s\]\s*(.+?)\s*$", re.MULTILINE)

def parse_todos(text):
    """Return list of unchecked todo item text from markdown."""
    return TODO_RE.findall(text or "")

def _open_items(proj):
    if not proj["todo_path"].exists():
        return []
    return parse_todos(proj["todo_path"].read_text(encoding="utf-8"))

def cmd_tasks(argv, root):
    p = argparse.ArgumentParser(prog="mem tasks")
    p.add_argument("name", nargs="?")
    args = p.parse_args(argv)
    projects = list(scan(root))
    if args.name:
        projects = [x for x in projects if x["name"] == args.name]
        if not projects:
            print(f"project not found: {args.name}", file=sys.stderr)
            return 1
    any_open = False
    for proj in sorted(projects, key=lambda x: x["name"].lower()):
        items = _open_items(proj)
        if not items:
            continue
        any_open = True
        print(f"[{proj['name']}]")
        for it in items:
            print(f"  - {it}")
    if not any_open:
        print("no open tasks")
    return 0

def cmd_active(argv, root):
    argparse.ArgumentParser(prog="mem active").parse_args(argv)
    active = [x for x in scan(root) if _open_items(x)]
    if not active:
        print("no active projects")
        return 0
    for proj in sorted(active, key=lambda x: x["name"].lower()):
        count = len(_open_items(proj))
        print(f"{proj['name']}  ({count} open)")
    return 0
```

```python
def cmd_search(argv, root):
    p = argparse.ArgumentParser(prog="mem search")
    p.add_argument("query")
    args = p.parse_args(argv)
    q = args.query.lower()
    matches = []
    for proj in scan(root):
        for fname in ("PROJECT.md", "todo.md"):
            fpath = proj["path"] / fname
            if not fpath.exists():
                continue
            try:
                lines = fpath.read_text(encoding="utf-8").splitlines()
            except (OSError, UnicodeDecodeError):
                continue
            for i, line in enumerate(lines, 1):
                if q in line.lower():
                    matches.append((proj["name"], fname, i, line.strip()))
    if not matches:
        print("no match")
        return 0
    for name, fname, i, line in matches:
        print(f"{name}:{fname}:{i}: {line}")
    return 0
```

```python
CANDIDATE_PARENTS = {"BUSINESS", "DATA", "PERSONAL", "PEOPLE"}
CANDIDATE_DEEP = {"CODE/CAMPAIGNS", "CODE/INFRA"}

def find_candidates(root):
    """Yield dirs under known parent dirs that have >=5 files and no PROJECT.md."""
    root = Path(root)
    for path in root.rglob("*"):
        if not path.is_dir():
            continue
        if any(part in EXCLUDED_DIRS for part in path.parts):
            continue
        rel = path.relative_to(root)
        parts = rel.parts
        if not parts:
            continue
        in_shallow = parts[0] in CANDIDATE_PARENTS and len(parts) == 2
        rel_str = "/".join(parts[:2])
        in_deep = rel_str in CANDIDATE_DEEP and len(parts) == 3
        if not (in_shallow or in_deep):
            continue
        if (path / "PROJECT.md").exists():
            continue
        try:
            file_count = sum(1 for _ in path.iterdir() if _.is_file())
        except OSError:
            continue
        if file_count < 5 and sum(1 for _ in path.iterdir() if _.is_dir()) < 3:
            continue
        yield path

def cmd_audit(argv, root):
    p = argparse.ArgumentParser(prog="mem audit")
    p.add_argument("--suggest", action="store_true")
    p.add_argument("--fill", action="store_true")
    args = p.parse_args(argv)
    cands = list(find_candidates(root))
    if not cands:
        print("no candidates found")
        return 0
    today = datetime.date.today().isoformat()
    for cand in cands:
        name = cand.name.lower().replace(" ", "-")
        if args.fill:
            stub_path = cand / "PROJECT.md"
            stub_path.write_text(PROJECT_STUB.format(name=name, today=today), encoding="utf-8")
            todo_path = cand / "todo.md"
            if not todo_path.exists():
                todo_path.write_text(TODO_STUB, encoding="utf-8")
            print(f"filled: {cand}")
        elif args.suggest:
            print(f"--- {cand} ---")
            print(PROJECT_STUB.format(name=name, today=today))
        else:
            print(cand)
    return 0
```

```python
    if not argv or argv[0] in ("-h", "--help"):
        print("usage: mem <command> [args]", file=sys.stderr)
        print("commands: " + ", ".join(COMMANDS.keys()), file=sys.stderr)

    cmd, rest = argv[0], argv[1:]
    if cmd not in COMMANDS:

    root = Path.cwd() if Path.cwd().resolve() == DEFAULT_ROOT.resolve() else DEFAULT_ROOT
    if not root.exists():
        root = Path.cwd()
    return COMMANDS[cmd](rest, root=root)
```
